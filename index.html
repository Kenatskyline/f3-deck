    const activeSuits = new Set(["C", "D", "H", "S"]);

    let drawPile = [];

    let drawHistory = [];

    let currentCard = null;


    const suitButtons = [...document.querySelectorAll(".suit-btn")];

    const duplicateInput = document.getElementById("duplicateCount");

    const cardShell = document.getElementById("cardShell");

    const frontImage = document.getElementById("frontImage");

    const drawBtn = document.getElementById("drawBtn");

    const shuffleBtn = document.getElementById("shuffleBtn");

    const undoBtn = document.getElementById("undoBtn");

    const statusText = document.getElementById("statusText");

    const cardGlow = document.getElementById("cardGlow");


    function sanitizeDuplicateCount() {

      let count = parseInt(duplicateInput.value, 10);

      if (Number.isNaN(count) || count < 1) count = 1;

      duplicateInput.value = count;

      return count;

    }


    function buildDeck() {

      const duplicateCount = sanitizeDuplicateCount();

      const nextDeck = [];


      for (const suit of allSuits) {

        if (!activeSuits.has(suit)) continue;

        for (const rank of ranks) {

          const code = `${rank}${suit}`;

          for (let i = 0; i < duplicateCount; i += 1) {

            nextDeck.push(code);

          }

        }

      }


      nextDeck.push("JOKER_1");

      nextDeck.push("JOKER_2");


      shuffleArray(nextDeck);

      return nextDeck;

    }


    function shuffleArray(arr) {

      for (let i = arr.length - 1; i > 0; i -= 1) {

        const j = Math.floor(Math.random() * (i + 1));

        [arr[i], arr[j]] = [arr[j], arr[i]];

      }

    }


    function getCallout(cardCode) {

      if (cardCode.startsWith("JOKER")) {

        return "Joker";

      }


      const rank = cardCode.slice(0, -1);

      const suit = cardCode.slice(-1);

      const spokenRank = rankNames[rank] || rank;

      const spokenSuit = suitNames[suit] || suit;

      const movement = movementMap[cardCode] || "Unknown movement";


      return `${spokenRank} of ${spokenSuit} — ${movement}`;

    }


    function speak(text) {

      if (!("speechSynthesis" in window)) return;

      window.speechSynthesis.cancel();

      const utterance = new SpeechSynthesisUtterance(text);

      utterance.rate = 0.96;

      utterance.pitch = 1;

      utterance.volume = 1;

      window.speechSynthesis.speak(utterance);

    }


    function setStatus(text) {

      statusText.textContent = text;

    }


    function setCardImage(cardCode) {

      frontImage.src = `cards/${cardCode}.png`;

      frontImage.alt = cardCode;

    }


    function showGlow() {

      cardGlow.classList.add("active");

      setTimeout(() => cardGlow.classList.remove("active"), 350);

    }


    function updateShufflePulse() {

      if (drawPile.length === 0) {

        shuffleBtn.classList.add("shuffle-alert");

      } else {

        shuffleBtn.classList.remove("shuffle-alert");

      }

    }


    function redrawCurrentCard() {

      if (!currentCard) {

        frontImage.src = "cards/BACK.png";

        cardShell.classList.remove("is-flipped");

        updateShufflePulse();

        return;

      }


      setCardImage(currentCard);

      cardShell.classList.add("is-flipped");

      updateShufflePulse();

    }


    function shuffleDeck() {

      if (activeSuits.size === 0) {

        setStatus("Select at least one suit.");

        return;

      }


      drawPile = buildDeck();

      drawHistory = [];

      currentCard = null;

      cardShell.classList.remove("is-flipped", "is-lifted");

      frontImage.src = "cards/BACK.png";

      setStatus(`Deck shuffled. ${drawPile.length} cards ready.`);

      updateShufflePulse();

    }


    function drawCard() {

      if (drawPile.length === 0) {

        setStatus("Deck empty. Shuffle to rebuild.");

        updateShufflePulse();

        return;

      }


      const nextCard = drawPile.pop();

      drawHistory.push({

        type: "draw",

        card: nextCard,

        previousCard: currentCard

      });

      currentCard = nextCard;


      cardShell.classList.remove("is-flipped", "is-lifted");


      setTimeout(() => {

        setCardImage(nextCard);

        cardShell.classList.add("is-flipped");

        requestAnimationFrame(() => {

          setTimeout(() => cardShell.classList.add("is-lifted"), 30);

        });

        const callout = getCallout(nextCard);

        setStatus(callout);

        showGlow();

        speak(callout);

        updateShufflePulse();

      }, 90);

    }


    function undoLastAction() {

      if (drawHistory.length === 0) {

        setStatus("Nothing to undo.");

        return;

      }


      const lastAction = drawHistory.pop();


      if (lastAction.type === "draw") {

        drawPile.push(lastAction.card);

        currentCard = lastAction.previousCard || null;


        cardShell.classList.remove("is-lifted");

        if (currentCard) {

          setCardImage(currentCard);

          cardShell.classList.add("is-flipped");

          setStatus(`Undo complete. Back to ${getCallout(currentCard)}.`);

        } else {

          frontImage.src = "cards/BACK.png";

          cardShell.classList.remove("is-flipped");

          setStatus("Undo complete.");

        }

      }


      updateShufflePulse();

    }


    function toggleSuit(suit, button) {

      if (activeSuits.has(suit)) {

        if (activeSuits.size === 1) {

          setStatus("At least one suit must stay enabled.");

          return;

        }

        activeSuits.delete(suit);

        button.classList.remove("active");

        button.classList.add("inactive");

      } else {

        activeSuits.add(suit);

        button.classList.add("active");

        button.classList.remove("inactive");

      }


      shuffleDeck();

    }


    suitButtons.forEach((button) => {

      button.addEventListener("click", () => {

        toggleSuit(button.dataset.suit, button);

      });

    });


    duplicateInput.addEventListener("change", () => {

      sanitizeDuplicateCount();

      shuffleDeck();

    });


    drawBtn.addEventListener("click", drawCard);

    shuffleBtn.addEventListener("click", shuffleDeck);

    undoBtn.addEventListener("click", undoLastAction);


    document.addEventListener("keydown", (event) => {

      if (event.code === "Space" || event.code === "Enter") {

        event.preventDefault();

        drawCard();

      }

    });


    shuffleDeck();
